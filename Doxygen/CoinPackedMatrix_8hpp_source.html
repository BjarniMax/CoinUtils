660  template < class FloatEqual >
661  bool
662  isEquivalent(const CoinPackedMatrix &rhs, const FloatEqual &eq) const
663  {
664  // Both must be column order or both row ordered and must be of same size
665  if ((isColOrdered() ^ rhs.isColOrdered()) || (getNumCols() != rhs.getNumCols()) || (getNumRows() != rhs.getNumRows()) || (getNumElements() != rhs.getNumElements()))
666  return false;
667 
668  for (int i = getMajorDim() - 1; i >= 0; --i) {
669  CoinShallowPackedVector pv = getVector(i);
670  CoinShallowPackedVector rhsPv = rhs.getVector(i);
671  if (!pv.isEquivalent(rhsPv, eq))
672  return false;
673  }
674  return true;
675  }
